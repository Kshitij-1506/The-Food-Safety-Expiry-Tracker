CREATE DATABASE food_safety_db;
USE food_safety_db;
CREATE TABLE users (
    user_id INT AUTO_INCREMENT PRIMARY KEY,
    username VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    role ENUM('Admin', 'Inspector', 'Developer', 'Viewer') NOT NULL
);
CREATE TABLE suppliers (
    supplier_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    contact VARCHAR(100),
    location VARCHAR(100)
);
CREATE TABLE products (
    product_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(100) NOT NULL,
    category VARCHAR(50),
    supplier_id INT,
    FOREIGN KEY (supplier_id) REFERENCES suppliers(supplier_id) ON DELETE CASCADE
);
CREATE TABLE batches (
    batch_id INT AUTO_INCREMENT PRIMARY KEY,
    product_id INT NOT NULL,
    manufacture_date DATE,
    expiry_date DATE,
    quantity INT,
    last_inspection_status VARCHAR(16) DEFAULT NULL,
    FOREIGN KEY (product_id) REFERENCES products(product_id) ON DELETE CASCADE
);
CREATE TABLE inspections (
    inspection_id INT AUTO_INCREMENT PRIMARY KEY,
    batch_id INT NOT NULL,
    inspector_name VARCHAR(100),
    inspection_date DATE,
    status ENUM('Pass', 'Fail') DEFAULT 'Pass',
    remarks TEXT,
    FOREIGN KEY (batch_id) REFERENCES batches(batch_id) ON DELETE CASCADE
);

DELIMITER //
CREATE PROCEDURE AddBatch(
    IN p_product_id INT,
    IN p_manufacture_date DATE,
    IN p_expiry_date DATE,
    IN p_quantity INT
)
BEGIN
    DECLARE productExists INT;
    SELECT COUNT(*) INTO productExists FROM products WHERE product_id = p_product_id;
    
    IF productExists = 1 THEN
        INSERT INTO batches (product_id, manufacture_date, expiry_date, quantity)
        VALUES (p_product_id, p_manufacture_date, p_expiry_date, p_quantity);
    ELSE
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Invalid Product ID — cannot create batch.';
    END IF;
END //
CREATE PROCEDURE AddInspection(
    IN p_batch_id INT,
    IN p_inspector_name VARCHAR(100),
    IN p_inspection_date DATE,
    IN p_status VARCHAR(10),
    IN p_remarks TEXT
)
BEGIN
    DECLARE batchExists INT;
    SELECT COUNT(*) INTO batchExists FROM batches WHERE batch_id = p_batch_id;
    
    IF batchExists = 1 THEN
        INSERT INTO inspections (batch_id, inspector_name, inspection_date, status, remarks)
        VALUES (p_batch_id, p_inspector_name, p_inspection_date, p_status, p_remarks);
    ELSE
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Invalid Batch ID — cannot add inspection.';
    END IF;
END //
DELIMITER ;

DELIMITER //

CREATE TRIGGER trg_update_batch_status
AFTER INSERT ON inspections
FOR EACH ROW
BEGIN
    UPDATE batches
    SET last_inspection_status = NEW.status
    WHERE batch_id = NEW.batch_id;
END //

DELIMITER ;
